33/** \brief Pack four floating point color values into a 32-bit integer form.
34
35 All of the color values should be between 0 and 1.
36
37 \param a Alpha value
38 \param r Red value
39 \param g Green value
40 \param b Blue value
41 \return The packed color value
42*/
43#define PVR_PACK_COLOR(a, r, g, b) ( \
44 ( ((uint8_t)( (a) * 255 ) ) << 24 ) | \
45 ( ((uint8_t)( (r) * 255 ) ) << 16 ) | \
46 ( ((uint8_t)( (g) * 255 ) ) << 8 ) | \
47 ( ((uint8_t)( (b) * 255 ) ) << 0 ) )
48
49/** \brief Pack two floating point coordinates into one 32-bit value,
50 truncating them to 16-bits each.
51
52 \param u First coordinate to pack
53 \param v Second coordinate to pack
54 \return The packed coordinates
55*/
56static inline uint32_t PVR_PACK_16BIT_UV(float u, float v) {
57 union {
58 float f;
59 uint32_t i;
60 } u2, v2;
61
62 u2.f = u;
63 v2.f = v;
64
65 return (u2.i & 0xFFFF0000) | (v2.i >> 16);
66}

89/** \brief Set cheap shadow parameters.
90 \ingroup pvr_global
91
92 This function sets up the PVR cheap shadow parameters for use. You can only
93 specify one scale value per frame, so the effect that you can get from this
94 is somewhat limited, but if you want simple shadows, this is the easiest way
95 to do it.
96
97 Polygons affected by a shadow modifier volume will effectively multiply
98 their final color by the scale value set here when shadows are enabled and
99 the polygon is inside the modifier (or outside for exclusion volumes).
100
101 \param enable Set to true to enable cheap shadow mode.
102 \param scale_value Floating point value (between 0 and 1) representing
103 how colors of polygons affected by and inside the
104 volume will be modified by the shadow volume.

183/** \defgroup pvr_txr_stride Texture Stride Management
184 \brief Configuration and retrieval of texture stride settings in VRAM.

186*/

188/** \brief Set the global stride width for non-power-of-two textures in PVR RAM.
189 \ingroup pvr_txr_stride
190
191 This function configures the global texture stride register
192 `PVR_TXR_STRIDE_MULT`, which defines the row width in VRAM for
193 non-power-of-two textures. The setting applies to all textures
194 rendered with the `PVR_TXRFMT_X32_STRIDE` flag in the same frame.
195 Since `PVR_TXR_STRIDE_MULT` is a global register, all textures
196 using this flag must share the same stride width in each frame.
197
198 The stride width configured here is **only supported for textures
199 with widths that are multiples of 32 pixels** and up to a maximum
200 of 992 pixels. Any texture width not meeting this requirement will
201 not work with the `PVR_TXRFMT_X32_STRIDE` flag.
202
203 \warning
204 - Textures that are palette-based cannot use the `PVR_TXRFMT_X32_STRIDE`
205 flag so the stride set here will not apply to them.
206
207 \param texture_width The width of the texture in pixels. Must be a
208 multiple of 32 and up to 992 pixels.
209 \retval true On success.
210 \retval false On failure.
211
212 \sa pvr_txr_get_stride()
213*/
214bool pvr_txr_set_stride(uint32_t texture_width);
215
216/** \brief Get the current texture stride width in pixels as set in the PVR.
217 \ingroup pvr_txr_stride
218
219 This function reads the `PVR_TXR_STRIDE_MULT` register and calculates the
220 texture stride width in pixels. The value returned is the width in pixels
221 that has been configured for all textures using the `PVR_TXRFMT_X32_STRIDE`
222 flag in the same frame.
223
224 The stride width is computed by taking the current multiplier in
225 `PVR_TXR_STRIDE_MULT` (which stores the width divided by 32), and
226 multiplying it back by 32 to return the full width in pixels.
227
228 \return The current texture stride width in pixels.
229
230 \sa pvr_txr_set_stride()
231*/
232uint32_t pvr_txr_get_stride(void);
